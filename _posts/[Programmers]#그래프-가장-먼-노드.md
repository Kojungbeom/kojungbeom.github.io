from collections import deque

def solution(n, edge):
    answer = 0
    queue = deque([[1, 0]])
    visited = [0]*(n+1)
    nodes = 0
    temp = []
    dict = {}
    
    for i in range(1, n+1):
        dict[i] = []
        
    for i in range(len(edge)):
        dict[edge[i][0]].append(edge[i][1])
        dict[edge[i][1]].append(edge[i][0])
    visited[1] =1
    while queue:
        n, d = queue.popleft()
        for i in dict[n]:
            if not visited[i]:
                queue.append([i, d+1])
                visited[i] = 1
        if not temp:
            temp.append([n, d])
        else:
            if d == temp[0][1]:
                temp.append([n,d])
            elif d > temp[0][1]:
                temp = []
                temp.append([n,d])        
    return len(temp)
